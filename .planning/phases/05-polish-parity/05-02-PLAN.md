### Task 1: 创建 use-keyboard-shortcuts hook

- src/components/workspace-shell.tsx（了解组件结构，确认哪些操作可以暴露给快捷键）

<action>
1. 创建 `src/hooks/use-keyboard-shortcuts.ts`：
   ```typescript
   'use client';

   import { useEffect } from 'react';

   type ShortcutActions = {
     onFocusInput: () => void;
     onNewSession: () => void;
     onPrevSession: () => void;
     onNextSession: () => void;
     onEscape: () => void;
   };

   export function useKeyboardShortcuts(actions: ShortcutActions) {
     useEffect(() => {
       const handler = (event: KeyboardEvent) => {
         const mod = event.metaKey || event.ctrlKey;

         // Cmd/Ctrl + K — 聚焦输入框
         if (mod && event.key === 'k') {
           event.preventDefault();
           actions.onFocusInput();
           return;
         }

         // Cmd/Ctrl + N — 新建会话
         if (mod && event.key === 'n') {
           event.preventDefault();
           actions.onNewSession();
           return;
         }

         // Cmd/Ctrl + [ — 上一个会话
         if (mod && event.key === '[') {
           event.preventDefault();
           actions.onPrevSession();
           return;
         }

         // Cmd/Ctrl + ] — 下一个会话
         if (mod && event.key === ']') {
           event.preventDefault();
           actions.onNextSession();
           return;
         }

         // Escape — 关闭展开/侧边栏
         if (event.key === 'Escape') {
           event.preventDefault();
           actions.onEscape();
           return;
         }
       };

       document.addEventListener('keydown', handler);
       return () => document.removeEventListener('keydown', handler);
     }, [actions]);
   }
   ```

2. 使用 `useCallback` 包装传入的 actions 对象以避免频繁重新绑定：
   - 在调用方（workspace-shell）中用 `useCallback` 包装每个 action
   - hook 的 `useEffect` 依赖数组包含 `[actions]`
</action>

3. 用 `useCallback` 包装快捷键 actions 并调用 hook：
   ```typescript
   const shortcutActions = useMemo(() => ({
     onFocusInput: () => chatInputRef.current?.focus(),
     onNewSession: () => void startNewSession(),
     onPrevSession: () => {
       if (!sessionList.length) return;
       const currentIndex = sessionList.findIndex(s => s.sessionId === sessionId);
       if (currentIndex > 0) {
         void switchSession(sessionList[currentIndex - 1].sessionId);
       }
     },
     onNextSession: () => {
       if (!sessionList.length) return;
       const currentIndex = sessionList.findIndex(s => s.sessionId === sessionId);
       if (currentIndex < sessionList.length - 1) {
         void switchSession(sessionList[currentIndex + 1].sessionId);
       }
     },
     onEscape: () => setIsSidebarOpen(false),
   }), [sessionList, sessionId, startNewSession, switchSession]);

   useKeyboardShortcuts(shortcutActions);
   ```

4. 将 `ChatInput` 组件加上 ref：
   ```tsx
   <ChatInput ref={chatInputRef} onSend={sendQuery} onAbort={abortSession} disabled={isStreaming} />
   ```
